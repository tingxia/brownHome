num_days = input("How many days of events would you like to hear about?")
category = input("What sort of events are you interested in?")

default_url = "http://events.brown.edu:/cal/main/listEvents.do?b=de&skinName=rss-list&setappvar=summaryMode(details)"

duration_filter = "&days=" + num_days
category_filter = "&cat=" + category

events = rss(default_url + duration_filter + category_filter)

for event in events:
  event_name = event[title]
  output(event_name)
